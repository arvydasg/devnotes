wsl commands

# View the list of distros and their current state:
wsl.exe -l -v
# Shutdown everything:
wsl.exe --shutdown
# Terminate a specific distro:
wsl.exe -t <DistroName>
# Boot up the default distro (marked with *):
wsl.exe
# Boot up a specific distro:
wsl.exe -d <DistroName>
